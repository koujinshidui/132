//gulp压缩工具指令
//定义一些任务,类似install
//npm install
var gulp = require('gulp');
//安装插件(别人的模块)npm install gulp-...;
//压缩混淆模块
var uglify = require('gulp-uglify');
//修改名字
var rename = require('gulp-rename');
//合并模块
var concat = require('gulp-concat');
//监听模块
var concat = require('gulp-update');

//1,打印一句话
//tast(任务名,任务的实现)任务命令;
//在命令中调用:gulp show;
gulp.task('show',function () {
    console.log("hello")
})

//2,复制
gulp.task('copy',function () {
    //复制lianxi.html到html中;
    gulp.src('lianxi.html').pipe(
        //路径不存在,会自动创建
        //存在html/中
        gulp.dest('html/')
    );

})

//3,压缩,混淆 安装插件
gulp.task('uglify',function () {
    //1,压缩 lianxi.html
    //2,压缩
    //3,新名字
    //4,新文件
    //压缩lianxi.js到js中;
    gulp.src('lianxi.js').pipe(
        uglify()
    ).pipe(
        rename(
            {
                suffix:'.min'
            }
        )
    ).pipe(
        gulp.dest('js/')
    )

});

//代码合并;
gulp.task('concat',function () {
    //被合并的文件
    gulp.src(['lianxi.js','lianxi1.js']).pipe(
        //合并concat(新文件名)
        concat('lianxi2.js')
    ).pipe(
        gulp.dest('js/')
    )
})


//6实时刷新,监听变化,
//1,webstorm jetbrains插件 webstrom debug运行
//2 gulp    browser-sync浏览器同步插件
gulp.task('update',function () {
    //1,初始化,服务器路径
    browser({
        server:{baseDir:'./'}
        }
    )
    //2,监听变化 刷新
    gulp.watch('lianxi.html',function () {
        //刷新浏览器
        browser.reload()
    })

});

//5一个命令 执行4个任务
gulp.task('all',['show','copy','uglify','concat','update']);
